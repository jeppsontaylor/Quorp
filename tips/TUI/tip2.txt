pub fn render_quorp_logo(width: usize, frame_time: f32, color: ColorCapability) -> Vec<String> {
    let logo = if width < 76 { COMPACT_LOGO } else { LOGO };

    logo.iter()
        .enumerate()
        .map(|(row, line)| {
            let centered = center_plain(line, width);
            if matches!(color, ColorCapability::NoColor) {

fn gradient_line(line: &str, row: usize, frame_time: f32) -> String {
    let mut out = String::new();
    let chars: Vec<char> = line.chars().collect();
    let denom = chars.len().saturating_sub(1).max(1) as f32;

    for (col, ch) in chars.into_iter().enumerate() {
        if ch == ' ' {
            out.push(ch);
            continue;
        }

        let wave = ((col as f32 / denom) + frame_time * 0.18 + row as f32 * 0.07) % 1.0;
        let rgb = if wave < 0.5 {
            lerp_rgb(SHIMMER_COOL, SHIMMER_WARM, wave * 2.0)
        } else {
            lerp_rgb(SHIMMER_WARM, ACCENT_YELLOW, (wave - 0.5) * 2.0)
        };

        out.push_str(&rgb.fg());
        out.push(ch);
    }

    out.push_str(RESET);
    out
}

fn paint(text: &str, color: ColorCapability, rgb: Rgb, bold: bool) -> String {
    if matches!(color, ColorCapability::NoColor) {
        return text.to_string();
    }

    if bold {
        format!("{}{}{}{}", rgb.fg(), BOLD, text, RESET)
    } else {
        format!("{}{}{}", rgb.fg(), text, RESET)
    }
}

fn center_plain(text: &str, width: usize) -> String {
    let text_width = text.width();
    let pad = width.saturating_sub(text_width) / 2;
    format!("{}{}", " ".repeat(pad), text)
}

fn center_styled(styled: &str, visible_width: usize, width: usize) -> String {
    let pad = width.saturating_sub(visible_width) / 2;
    format!("{}{}", " ".repeat(pad), styled)
}

fn truncate_plain(value: &str, max_width: usize) -> String {
    if value.width() <= max_width {
        return value.to_string();
    }

    if max_width <= 1 {
        return "…".to_string();
    }

    let target = max_width - 1;
    let mut out = String::new();
    let mut width = 0;

    for ch in value.chars() {
        let ch_width = unicode_width::UnicodeWidthChar::width(ch).unwrap_or(0);
        if width + ch_width > target {
            break;
        }
        out.push(ch);
        width += ch_width;
    }

    out.push('…');
    out
}

pub fn render_shell_scene(
    scene: &ShellScene,
    width: usize,
    height: usize,
    color: ColorCapability,

    let width = width.max(56);
    let height = height.max(10);

    let mut body = Vec::new();

    if scene.show_logo && height >= 22 {
        body.extend(render_quorp_logo(width, scene.frame_time, color));
        body.push(String::new());
        body.push(center_styled(
            &paint("Ask anything. Use / for commands.", color, ACCENT_YELLOW, false),

            width,
        ));
        body.push(String::new());
    }

    for line in &scene.transcript {
        body.extend(render_transcript_row(line, width, color));
    }

    if let Some(active) = scene.active.as_ref() {
        if !matches!(active.state, CommandState::Pending) {
            if !body.is_empty() {
                body.push(String::new());
            }
            body.extend(render_work_card(active, width, color));
        }
    }

    let suggestion_lines = render_suggestions(
        &scene.suggestions,
        scene.selected_suggestion,
        width,
        color,
    );

    let composer = render_composer(&scene.composer, width, color);
    let footer = render_footer(&scene.footer, width, color);

    let reserved = suggestion_lines.len() + composer.lines.len() + 1;
    let body_height = height.saturating_sub(reserved);

    let visible_body = visible_window(&body, body_height, scene.scroll_offset);

    let mut lines = Vec::new();
    lines.extend(visible_body);

    while lines.len() < body_height {
        lines.push(String::new());
    }

    lines.extend(suggestion_lines);

    let cursor_row = lines.len().saturating_add(1).min(u16::MAX as usize) as u16;
    let cursor_col = composer.cursor_col;

    lines.extend(composer.lines);
    lines.push(footer);

    ShellSceneRenderOutput {
        lines,
        cursor_row,
        cursor_col,
    }
}

struct ComposerRender {
    lines: Vec<String>,
    cursor_col: u16,
}

fn render_composer(buffer: &str, width: usize, color: ColorCapability) -> ComposerRender {
    let plain = matches!(color, ColorCapability::NoColor);
    let (tl, tr, bl, br, h, v) = if plain {
        ("+", "+", "+", "+", "-", "|")
    } else {
        ("╭", "╮", "╰", "╯", "─", "│")
    };

    let border_color = ACCENT_VIOLET;
    let inner = width.saturating_sub(4);
    let prompt = if plain { "> " } else { "❯ " };
    let input_width = inner.saturating_sub(prompt.width());

    let shown = tail_to_width(buffer, input_width);
    let shown_width = strip_ansi_width(&shown);
    let padding = inner.saturating_sub(prompt.width() + shown_width);

    let mut lines = Vec::new();

    lines.push(format!(
        "{}{}{}",
        paint(tl, color, border_color, false),
        paint(&h.repeat(width.saturating_sub(2)), color, border_color, false),
        paint(tr, color, border_color, false)
    ));

    let prompt_styled = paint(prompt, color, ACCENT_CYAN, true);
    lines.push(format!(
        "{} {}{}{} {}",
        paint(v, color, border_color, false),
        prompt_styled,
        paint(&shown, color, FG_TEXT, false),
        " ".repeat(padding),
        paint(v, color, border_color, false),
    ));

    lines.push(format!(
        "{}{}{}",
        paint(bl, color, border_color, false),
        paint(&h.repeat(width.saturating_sub(2)), color, border_color, false),
        paint(br, color, border_color, false)
    ));

    ComposerRender {
        lines,
        cursor_col: (2 + prompt.width() + shown_width)
            .min(u16::MAX as usize) as u16,
    }
}

fn render_transcript_row(line: &ShellLine, width: usize, color: ColorCapability) -> Vec<String> {
    let plain = matches!(color, ColorCapability::NoColor);

    let (glyph, rgb, text_rgb, dim_text) = match line.kind {
        ShellLineKind::Hint => ("·", ACCENT_VIOLET, FG_TEXT, true),
        ShellLineKind::User => ("›", ACCENT_CYAN, FG_TEXT, false),
        ShellLineKind::Assistant => ("✦", ACCENT_GREEN, FG_TEXT, false),
        ShellLineKind::Tool => ("⎿", ACCENT_YELLOW, FG_TEXT, true),
        ShellLineKind::Status => ("•", ACCENT_VIOLET, FG_TEXT, true),
        ShellLineKind::Warn => ("!", ACCENT_RED, ACCENT_RED, false),
    };

    let prefix = format!("  {glyph} ");
    let continuation_prefix = "    ";
    let first_content_width = width.saturating_sub(prefix.width()).max(8);
    let cont_content_width = width.saturating_sub(continuation_prefix.width()).max(8);
    let chunks = wrap_plain(&line.text, first_content_width, cont_content_width);

    let mut out = Vec::new();

    for (index, chunk) in chunks.iter().enumerate() {
        if index == 0 {
            if plain {
                out.push(format!("{prefix}{chunk}"));
            } else {
                let text = if dim_text {
                    format!("{DIM}{}{RESET}", chunk)
                } else {
                    paint(chunk, color, text_rgb, false)
                };
                out.push(format!(
                    "  {} {}",
                    paint(glyph, color, rgb, true),
                    text
                ));
            }
        } else if plain {
            out.push(format!("{continuation_prefix}{chunk}"));
        } else {
            let text = if dim_text {
                format!("{DIM}{}{RESET}", chunk)
            } else {
                paint(chunk, color, text_rgb, false)
            };
            out.push(format!("{continuation_prefix}{text}"));
        }
    }

    if out.is_empty() {
        out.push(prefix);
    }

    out
}

fn render_work_card(card: &CommandCard, width: usize, color: ColorCapability) -> Vec<String> {
    let plain = matches!(color, ColorCapability::NoColor);
    let (tl, mid, bl, v) = if plain {
        ("+-", "| ", "+-", "| ")
    } else {
        ("╭─", "│ ", "╰─", "│ ")
    };

    let border_color = match card.state {
        CommandState::Pending => ACCENT_VIOLET,
        CommandState::Active { .. } => ACCENT_CYAN,
        CommandState::Passed { .. } => ACCENT_GREEN,
        CommandState::Failed { .. } => ACCENT_RED,
    };

    let state = render_command_state(&card.state, color);
    let usable = width.saturating_sub(4).max(20);

    let mut lines = Vec::new();

    lines.push(format!(
        "  {} {} {}",
        paint(tl, color, border_color, false),
        paint(&truncate_plain(&card.label, 24), color, FG_TEXT, true),
        state,
    ));

    lines.push(format!(
        "  {}{}",
        paint(mid, color, border_color, false),
        paint(
            &truncate_plain(&card.command, usable.saturating_sub(2)),
            color,
            FG_TEXT,
            false
        ),
    ));

    if !card.cwd.trim().is_empty() {
        lines.push(format!(
            "  {}{}",
            paint(mid, color, border_color, false),
            paint(
                &truncate_plain(&format!("cwd {}", card.cwd), usable.saturating_sub(2)),
                color,
                ACCENT_VIOLET,
                false
            ),
        ));
    }

    if !card.output_summary.trim().is_empty() {
        lines.push(format!(
            "  {}{}",
            paint(mid, color, border_color, false),
            paint(
                &truncate_plain(&card.output_summary, usable.saturating_sub(2)),
                color,
                FG_TEXT,
                false
            ),
        ));
    }

    lines.push(format!("  {}", paint(bl, color, border_color, false)));
    lines
}

fn render_command_state(state: &CommandState, color: ColorCapability) -> String {
    match state {
        CommandState::Pending => paint("ready", color, ACCENT_VIOLET, false),
        CommandState::Active { frame_time } => {
            let spinner = match ((*frame_time * 10.0) as usize) % 8 {
                0 => "⠋",
                1 => "⠙",
                2 => "⠹",
                3 => "⠸",
                4 => "⠼",
                5 => "⠴",
                6 => "⠦",
                _ => "⠧",
            };
            paint(&format!("{spinner} running"), color, ACCENT_CYAN, false)
        }
        CommandState::Passed { exit_code, duration } => {
            paint(&format!("✓ exit={exit_code} {duration}"), color, ACCENT_GREEN, false)
        }
        CommandState::Failed { exit_code, duration } => {
            paint(&format!("✗ exit={exit_code} {duration}"), color, ACCENT_RED, false)
        }
    }
}

fn render_suggestions(
    suggestions: &[ShellSuggestion],
    selected: usize,
    width: usize,
    color: ColorCapability,
) -> Vec<String> {
    suggestions
        .iter()
        .take(7)
        .enumerate()
        .map(|(index, entry)| {
            let selected_row = index == selected;
            let selector = if selected_row { "›" } else { " " };

            let value = truncate_plain(&entry.value, 18);
            let detail = truncate_plain(&entry.detail, 14);
            let remaining = width.saturating_sub(28);
            let description = truncate_plain(&entry.description, remaining);

            if matches!(color, ColorCapability::NoColor) {
                format!(" {selector} {value:<18} {detail:<14} {description}")
            } else if selected_row {
                format!(
                    " {} {}{}{} {}{}{} {}",
                    paint(selector, color, ACCENT_YELLOW, true),
                    ACCENT_CYAN.fg(),
                    value,
                    RESET,
                    ACCENT_VIOLET.fg(),
                    detail,
                    RESET,
                    paint(&description, color, FG_TEXT, false)
                )
            } else {
                format!(
                    "   {} {} {}",
                    paint(&format!("{value:<18}"), color, ACCENT_CYAN, false),
                    paint(&format!("{detail:<14}"), color, ACCENT_VIOLET, false),
                    paint(&description, color, FG_TEXT, false)
                )
            }
        })
        .collect()
}

fn render_footer(footer: &ShellFooter, width: usize, color: ColorCapability) -> String {
    let plain = matches!(color, ColorCapability::NoColor);
    let text = format!(
        "{}  {}  {}  {}",
        footer.model_provider, footer.mode_label, footer.phase, footer.usage
    );
    let text = truncate_plain(&text, width);

    if plain {
        return text;
    }

    let mut out = String::new();
    out.push_str(&ACCENT_CYAN.fg());
    out.push_str(&truncate_plain(&footer.model_provider, 34));
    out.push_str(RESET);
    out.push_str("  ");

    out.push_str(&ACCENT_YELLOW.fg());
    out.push_str(&truncate_plain(&footer.mode_label, 24));
    out.push_str(RESET);
    out.push_str("  ");

    out.push_str(&ACCENT_VIOLET.fg());
    out.push_str(&truncate_plain(&footer.phase, 28));
    out.push_str(RESET);
    out.push_str("  ");

    let used = strip_ansi_width(&out);
    let remaining = width.saturating_sub(used);
    out.push_str(&paint(
        &truncate_plain(&footer.usage, remaining),
        color,
        FG_TEXT,
        false,
    ));

    truncate_ansiish(&out, width)
}

fn visible_window(lines: &[String], height: usize, scroll_offset: usize) -> Vec<String> {
    if height == 0 {
        return Vec::new();
    }

    if lines.len() <= height {
        return lines.to_vec();
    }

    if scroll_offset == 0 {
        return lines[lines.len().saturating_sub(height)..].to_vec();
    }

    let max_scroll = lines.len().saturating_sub(height);
    let scroll_offset = scroll_offset.min(max_scroll);
    let end = lines.len().saturating_sub(scroll_offset);
    let start = end.saturating_sub(height);

    lines[start..end].to_vec()
}

fn wrap_plain(text: &str, first_width: usize, continuation_width: usize) -> Vec<String> {
    let mut chunks = Vec::new();
    let mut current = String::new();
    let mut current_width = 0usize;
    let mut limit = first_width.max(1);

    for word in text.split_whitespace() {
        let word_width = word.width();
        let spacer = usize::from(!current.is_empty());

        if current_width + spacer + word_width > limit && !current.is_empty() {
            chunks.push(current);
            current = String::new();
            current_width = 0;
            limit = continuation_width.max(1);
        }

        if !current.is_empty() {
            current.push(' ');
            current_width += 1;
        }

        if word_width > limit {
            for ch in word.chars() {
                let ch_width = ch.width().unwrap_or(0);
                if current_width + ch_width > limit && !current.is_empty() {
                    chunks.push(current);
                    current = String::new();
                    current_width = 0;
                    limit = continuation_width.max(1);
                }
                current.push(ch);
                current_width += ch_width;
            }
        } else {
            current.push_str(word);
            current_width += word_width;
        }
    }

    if !current.is_empty() {
        chunks.push(current);
    }

    if chunks.is_empty() {
        chunks.push(String::new());
    }

    chunks
}

fn tail_to_width(text: &str, max_width: usize) -> String {
    if text.width() <= max_width {
        return text.to_string();
    }

    if max_width <= 1 {
        return "…".to_string();
    }

    let target = max_width - 1;
    let mut out = String::new();
    let mut width = 0usize;

    for ch in text.chars().rev() {
        let ch_width = ch.width().unwrap_or(0);
        if width + ch_width > target {
            break;
        }
        out.insert(0, ch);
        width += ch_width;
    }

    format!("…{out}")
}

fn truncate_plain(text: &str, max_width: usize) -> String {
    if text.width() <= max_width {
        return text.to_string();
    }

    if max_width <= 1 {
        return "…".to_string();
    }

    let target = max_width - 1;
    let mut out = String::new();
    let mut width = 0usize;

    for ch in text.chars() {
        let ch_width = ch.width().unwrap_or(0);
        if width + ch_width > target {
            break;
        }
        out.push(ch);
        width += ch_width;
    }

    out.push('…');
    out
}

fn truncate_ansiish(text: &str, max_width: usize) -> String {
    if strip_ansi_width(text) <= max_width {
        return text.to_string();
    }

    // Conservative fallback: strip styling only when truncating a complex line.
    truncate_plain(&strip_ansi(text), max_width)
}

fn strip_ansi(text: &str) -> String {
    let mut out = String::new();
    let mut in_escape = false;

    for ch in text.chars() {
        if in_escape {
            if ch == 'm' {
                in_escape = false;
            }
            continue;
        }

        if ch == '\x1b' {
            in_escape = true;
            continue;
        }

        out.push(ch);
    }

    out
}

fn strip_ansi_width(text: &str) -> usize {
    strip_ansi(text).width()
}

fn paint(text: &str, color: ColorCapability, rgb: Rgb, bold: bool) -> String {
    if matches!(color, ColorCapability::NoColor) {
        return text.to_string();
    }

    if bold {
        format!("{}{}{}{}", rgb.fg(), BOLD, text, RESET)
    } else {
        format!("{}{}{}", rgb.fg(), text, RESET)
    }
}

fn center_styled(styled: &str, visible_width: usize, width: usize) -> String {
    let pad = width.saturating_sub(visible_width) / 2;
    format!("{}{}", " ".repeat(pad), styled)
}
